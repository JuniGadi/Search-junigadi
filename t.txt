    let selectedCategory = "";
    let selectedPrice = "";
    let selectedLocation = "";

    const vehicleTypeContainer = document.getElementById("vehicleType");
    const priceRangeContainer = document.getElementById("priceRange");
    const locationContainer = document.getElementById("location");
    const moreLocationsContainer = document.getElementById("moreLocations");

    // Vehicle type selection
    vehicleTypeButtons.forEach((button) => {
      button.addEventListener("click", () => {
        toggleSelected(vehicleTypeButtons, button);
        selectedCategory = button.getAttribute("data-type");
        showStep(priceRangeContainer);
        hideStep(vehicleTypeContainer);
        updateProgress(66.66);
        updateSearchButtonVisibility();
      });
    });

    // Price range selection
    priceRangeButtons.forEach((button) => {
      button.addEventListener("click", () => {
        toggleSelected(priceRangeButtons, button);
        selectedPrice = button.getAttribute("data-price");
        showStep(locationContainer);
        hideStep(priceRangeContainer);
        updateProgress(100);
        updateSearchButtonVisibility();
      });
    });

    // Location selection
    locationButtons.forEach((button) => {
      button.addEventListener("click", () => {
        toggleSelected(locationButtons, button);
        selectedLocation = button.getAttribute("data-location");
        updateSearchButtonVisibility();
      });
    });

    // More locations button
    moreLocationsButton.addEventListener("click", () => {
      showStep(moreLocationsContainer);
      hideStep(locationContainer);
    });

    // Dropdown selection
    locationDropdown.addEventListener("change", () => {
      selectedLocation = locationDropdown.value;
      updateSearchButtonVisibility();
    });

    // Back button functionality
    backButtons.forEach((button) => {
      button.addEventListener("click", () => {
        const currentStep = button.closest('.step');
        
        if (currentStep.id === 'priceRange') {
          showStep(vehicleTypeContainer);
          hideStep(priceRangeContainer);
          selectedCategory = "";
          clearSelections(vehicleTypeButtons);
          updateProgress(33.33);
        } else if (currentStep.id === 'location') {
          showStep(priceRangeContainer);
          hideStep(locationContainer);
          selectedPrice = "";
          clearSelections(priceRangeButtons);
          updateProgress(66.66);
        } else if (currentStep.id === 'moreLocations') {
          showStep(locationContainer);
          hideStep(moreLocationsContainer);
          locationDropdown.value = "";
        }
        
        selectedLocation = "";
        updateSearchButtonVisibility();
      });
    });

    // Search button functionality
    searchButton.addEventListener("click", (event) => {
      if (!selectedCategory || !selectedPrice) {
        showNotification("Please select both type of vehicle and price range.");
        event.preventDefault();
        return;
      }
      const code = getCode(selectedCategory, selectedPrice);
      const locationQuery = selectedLocation ? `%20${selectedLocation}` : "";
      const link = `https://mydukaan.io/junigadi/search?query=${code}${locationQuery}`;
      searchButton.href = link;
    });

    function showStep(step) {
      step.classList.remove('hidden');
    }

    function hideStep(step) {
      step.classList.add('hidden');
    }

    function toggleSelected(buttons, selectedButton) {
      buttons.forEach(btn => btn.classList.remove('selected'));
      selectedButton.classList.add('selected');
    }

    function clearSelections(buttons) {
      buttons.forEach(btn => btn.classList.remove('selected'));
    }

    function updateProgress(percentage) {
      progressFill.style.width = percentage + '%';
    }

    function updateSearchButtonVisibility() {
      if (selectedCategory && selectedPrice) {
        searchButton.classList.remove("hidden");
      } else {
        searchButton.classList.add("hidden");
      }
    }

    function showNotification(message) {
      // Create a modern notification
      const notification = document.createElement('div');
      notification.style.cssText = `
        position: fixed;
        top: 20px;
        left: 50%;
        transform: translateX(-50%);
        background: linear-gradient(135deg, #667eea 0%, #764ba2 100%);
        color: white;
        padding: 1rem 2rem;
        border-radius: 12px;
        box-shadow: 0 10px 30px rgba(0,0,0,0.2);
        z-index: 1000;
        font-weight: 600;
        animation: slideDown 0.3s ease-out;
      `;
      notification.textContent = message;
      
      const style = document.createElement('style');
      style.textContent = `
        @keyframes slideDown {
          from { opacity: 0; transform: translateX(-50%) translateY(-20px); }
          to { opacity: 1; transform: translateX(-50%) translateY(0); }
        }
      `;
      document.head.appendChild(style);
      document.body.appendChild(notification);
      
      setTimeout(() => {
        notification.remove();
        style.remove();
      }, 3000);
    }
